// lib/location/validateCoords.ts // Координатын бүрэн шалгалт — нэг газраас /** Монгол улсын хилийн хүрээ */ const MONGOLIA_BOUNDS = { latMin: 41.5, latMax: 52.2, lngMin: 87.7, lngMax: 119.9, } /** Улаанбаатар хотын хүрээ (нарийн шалгалтад) */ const UB_BOUNDS = { latMin: 47.7, latMax: 48.1, lngMin: 106.6, lngMax: 107.3, } export type CoordStatus = | 'valid_ub' // УБ хотод байна | 'valid_mongolia' // Монголд байна, УБ-с гадна | 'outside' // Монголоос гадна | 'zero' // 0,0 — хуучин placeholder | 'missing' // null / undefined / NaN export interface CoordValidation { status: CoordStatus valid: boolean // газрын зурагт харуулах эсэх message: string // хэрэглэгчид харуулах текст needsUpdate: boolean // дэлгүүр эзэнд сануулах эсэх } export function validateCoords( lat: number | null | undefined, lng: number | null | undefined ): CoordValidation { // 1. Байхгүй / NaN if (lat == null || lng == null || isNaN(lat) || isNaN(lng)) { return { status: 'missing', valid: false, message: 'Байршлын координат оруулаагүй байна. Газрын зураг дээр pin тавина уу.', needsUpdate: true, } } // 2. 0,0 — хуучин өгөгдлийн placeholder if (lat === 0 && lng === 0) { return { status: 'zero', valid: false, message: 'Байршил тогтоогоогүй байна (0, 0). Газрын зурагт тэмдэглэнэ үү.', needsUpdate: true, } } // 3. Монголоос гадна const inMongolia = ( lat >= MONGOLIA_BOUNDS.latMin && lat <= MONGOLIA_BOUNDS.latMax && lng >= MONGOLIA_BOUNDS.lngMin && lng <= MONGOLIA_BOUNDS.lngMax ) if (!inMongolia) { return { status: 'outside', valid: false, message: `Координат Монголоос гадна байна (${lat.toFixed(4)}, ${lng.toFixed(4)}). Шалгана уу.`, needsUpdate: true, } } // 4. УБ хотод const inUB = ( lat >= UB_BOUNDS.latMin && lat <= UB_BOUNDS.latMax && lng >= UB_BOUNDS.lngMin && lng <= UB_BOUNDS.lngMax ) if (inUB) { return { status: 'valid_ub', valid: true, message: 'Улаанбаатар хотод байна', needsUpdate: false, } } // 5. Монголд байна, УБ-с гадна (аймгийн төв гэх мэт) return { status: 'valid_mongolia', valid: true, message: 'Монгол улсад байна', needsUpdate: false, } } /** Массив шалгах (admin bulk check-д) */ export function validateCoordsMany( locations: Array<{ id: string; name: string; lat: number | null; lng: number | null }> ): Array<{ id: string; name: string; validation: CoordValidation }> { return locations.map(loc => ({ id: loc.id, name: loc.name, validation: validateCoords(loc.lat, loc.lng), })) } /** Статусын тоо гаргах */ export function getCoordStats(locations: Array<{ lat: number | null; lng: number | null }>) { const results = locations.map(l => validateCoords(l.lat, l.lng)) return { total: results.length, valid: results.filter(r => r.valid).length, missing: results.filter(r => r.status === 'missing').length, zero: results.filter(r => r.status === 'zero').length, outside: results.filter(r => r.status === 'outside').length, needsUpdate:results.filter(r => r.needsUpdate).length, } }
