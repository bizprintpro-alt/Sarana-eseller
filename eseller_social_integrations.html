## FACEBOOK SHOP CATALOG SYNC Automatically sync eseller.mn products to Facebook Commerce Manager (Facebook Shop / Marketplace) using the Catalog API. --- ### PRISMA SCHEMA ```prisma model FbCatalogSync { id String @id @default(cuid()) entityId String // store/company/agent id fbCatalogId String // FB catalog ID fbPageId String status SyncStatus @default(IDLE) totalItems Int @default(0) syncedItems Int @default(0) failedItems Int @default(0) lastSyncAt DateTime? nextSyncAt DateTime? errorLog Json? createdAt DateTime @default(now()) } model ProductFbMapping { id String @id @default(cuid()) productId String @unique fbRetailerId String @unique // FB product retailer_id (= eseller refId) fbItemId String? // FB assigned item id lastSyncedAt DateTime? syncStatus String // "synced" | "pending" | "failed" | "deleted" errorMessage String? } enum SyncStatus { IDLE | RUNNING | COMPLETED | FAILED | PAUSED } ``` ### CATALOG SERVICE ```typescript // lib/integrations/facebook/CatalogService.ts class FacebookCatalogService { private graphUrl = "https://graph.facebook.com/v19.0" // Create catalog for entity (one-time setup) async createCatalog(entityId: string): Promise { const conn = await getEntityFbConnection(entityId) const res = await this.call(`/${conn.fbPageId}/owned_product_catalogs`, { method: 'POST', body: { name: `${entity.name} — Eseller Catalog`, vertical: 'commerce', }, token: decrypt(conn.pageToken), }) await db.fbCatalogSync.create({ data: { entityId, fbCatalogId: res.id, fbPageId: conn.fbPageId } }) return res.id } // Map eseller product to FB catalog item format mapProductToFbItem(product: Product): FbCatalogItem { return { id: product.refId, // retailer_id title: product.name, description: product.description?.slice(0, 9999) || "", availability: product.stock > 0 ? 10).join(","), brand: product.store?.name || "Eseller", google_product_category: mapCategoryToGoogle(product.category), custom_label_0: product.tier, // VIP | FEATURED | NORMAL custom_label_1: product.category.name, } } // Batch upload products (max 1000 per request) async batchSync(entityId: string, products: Product[]): Promise { const catalog = await db.fbCatalogSync.findUnique({ where: { entityId } }) const chunks = chunkArray(products, 1000) let synced = 0, failed = 0 const errors = [] for (const chunk of chunks) { const requests = chunk.map(p => ({ method: "UPDATE", retailer_id: this.getPageToken(entityId), }) // Check individual item results for (const handle of res.handles) { if (handle.errors?.length) { failed++ errors.push({ retailer_id: handle.retailer_id, error: handle.errors[0] }) } else { synced++ } } } catch (err) { failed += chunk.length errors.push({ batch_error: err.message }) } } await db.fbCatalogSync.update({ where: { entityId }, data: { syncedItems: "UPDATE", retailer_id: product.refId, data: this.mapProductToFbItem(product) }], }, token: await this.getPageToken(entityId), }) await db.productFbMapping.upsert({ where: { productId }, create: { productId, fbRetailerId: product.refId, lastSyncedAt: new Date(), syncStatus: 'synced' }, update: { lastSyncedAt: new Date(), syncStatus: 'synced' }, }) } // Delete product from FB catalog async Date() }, }) } } // Hook into product lifecycle: // After createProduct → catalogService.syncProduct(product.id) // After updateProduct → catalogService.syncProduct(product.id) // After deleteProduct
