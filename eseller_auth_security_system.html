## PASSWORDLESS AUTH SYSTEM — SMS OTP + Biometric + Passkey + DAN Build a complete passwordless authentication system for eseller.mn. Primary: SMS OTP (like Zary.mn). Progressive upgrade to biometric/passkey. --- ### 1. AUTH FLOW ARCHITECTURE ``` FIRST LOGIN / REGISTER: Phone → SMS OTP (6 digit, 5 min) → Profile setup → JWT issued SUBSEQUENT LOGINS (web): Phone → SMS OTP (cached 30 days on trusted device) SUBSEQUENT LOGINS (mobile app): Biometric (FaceID/TouchID) → JWT from secure refresh token ENTITY ACCOUNTS (store/agent/company): Same as above + entity selection after auth STEP-UP (sensitive actions): Re-verify via SMS OTP regardless of session age ``` --- ### 2. SMS OTP SERVICE ```typescript // lib/auth/OTPService.ts import argon2 from 'argon2' import { createClient } from 'redis' class OTPService { private redis = createClient({ url: process.env.REDIS_URL }) // Generate and send OTP async send(phone: string, purpose: OTPPurpose, ip: string): Promise { // Rate limit: max 3 OTP per phone per 10 minutes const rateLimitKey = `otp_rate:${phone}` const count = await this.redis.incr(rateLimitKey) if (count === 1) await this.redis.expire(rateLimitKey, 600) if (count > 3) throw new TooManyRequestsError( 'Хэт олон хүсэлт. 10 минутын дараа дахин оролдоно уу.' ) // Generate cryptographically secure 6-digit OTP const otp = crypto.randomInt(100000, 999999).toString() const expiresAt = new Date(Date.now() + 5 * 60 * 1000) // 5 min // Hash before storing — NEVER store plaintext OTP const codeHash = await argon2.hash(otp, { type: argon2.argon2id }) await db.oTPCode.create({ data: { phone, codeHash, purpose, expiresAt, ipAddress: ip, } }) // Send via SMS gateway // Primary: Unitel MobiFinance API // Fallback: MobiCom SMS API await smsGateway.send({ to: phone, message: `Eseller нэвтрэх код: ${otp}\n5 минутад хүчинтэй. Бусдад хэлэх хэрэггүй!`, }) } // Verify OTP async verify(phone: string, code: string, purpose: OTPPurpose): Promise { const record = await db.oTPCode.findFirst({ where: { phone, purpose, expiresAt: { gt: new Date() }, usedAt: null, attempts: { lt: 5 }, }, orderBy: { createdAt: 'desc' }, }) if (!record) throw new OTPExpiredError('Код хүчинтэй хугацаа дууссан эсвэл олдсонгүй.') // Increment attempt count await db.oTPCode.update({ where: { id: record.id }, data: { attempts: { increment: 1 } } }) const isValid = await argon2.verify(record.codeHash, code) if (!isValid) { if (record.attempts + 1 >= 5) { throw new OTPLockedError('5 удаа буруу код оруулсан. Шинэ код хүс.') } throw new OTPInvalidError(`Код буруу байна. ${4 - record.attempts} оролдлого үлдсэн.`) } // Mark as used (one-time use) await db.oTPCode.update({ where: { id: record.id }, data: { usedAt: new Date() } }) return true } } ``` --- ### 4. STEP-UP AUTHENTICATION MIDDLEWARE ```typescript // lib/security/StepUpAuth.ts const STEP_UP_RULES: Record = { // Capture: require re-verify if session older than threshold 'payment.create': { maxSessionAge: 30 * 60, // 30 минут threshold: 0, // Бүх дүнд methods: ['OTP'], }, 'payment.high_value': { maxSessionAge: 5 * 60, // 5 минут threshold: 50_000, // 50,000₮+ methods: ['OTP', 'BIOMETRIC'], }, 'payment.method.add': { maxSessionAge: 0, // Заавал шинэ verify threshold: 0, methods: ['OTP', 'BIOMETRIC'], }, 'withdrawal.request': { maxSessionAge: 0, threshold: 0, methods: ['OTP'], cooldown: 60 * 60, // Сүүлийн 1 цагт зөвхөн 1 удаа }, 'account.delete': { maxSessionAge: 0, methods: ['OTP'], additionalDelay: 24 * 60 * 60, // 24 цагийн cooling period }, 'seller.payout': { maxSessionAge: 0, threshold: 0, methods: ['OTP'], adminApprovalRequired: true, // 100,000₮+ }, } // Middleware для sensitive routes export function requireStepUp(action: string) { return async (req: Request) => { const rule = STEP_UP_RULES[action] if (!rule) return // no rule = no step-up required const session = await getSession(req) const sessionAge = (Date.now() - session.lastActiveAt.getTime()) / 1000 if (sessionAge > rule.maxSessionAge) { throw new StepUpRequiredError({ action, methods: rule.methods, message: 'Аюулгүй байдлын үүднээс дахин баталгаажуулна уу.', }) } // Check amount threshold const amount = req.body?.amount || 0 if (rule.threshold > 0 && amount >= rule.threshold) { // Check if step-up was completed for this session const stepUpKey = `stepup:${session.userId}:${action}` const completed = await redis.get(stepUpKey) if (!completed) { throw new StepUpRequiredError({ action, methods: rule.methods }) } } } } ``` ### 2. FRAUD DETECTION ```typescript // lib/security/FraudDetector.ts interface RiskSignal { type: string score: number // 0-100, higher = more risky reason: string } class FraudDetector { async assess(userId: string, action: string, context: Record): Promise { const signals: RiskSignal[] = [] // Signal 1: New device const deviceAge = await this.getDeviceAge(userId, context.deviceId as string) if (deviceAge < 24 * 60 * 60) { signals.push({ type: 'new_device', score: 30, reason: 'Шинэ төхөөрөмж' }) } // Signal 2: New location (different city/country) const locationChange = await this.detectLocationChange(userId, context.ip as string) if (locationChange.isDifferentCountry) { signals.push({ type: 'geo_anomaly', score: 60, reason: `Шинэ улс: ${locationChange.country}` }) } else if (locationChange.isDifferentCity) { signals.push({ type: 'geo_soft', score: 20, reason: `Шинэ хот: ${locationChange.city}` }) } // Signal 3: Velocity — too many payments in short time const recentPayments = await db.order.count({ where: { buyerId: userId, createdAt: { gt: subHours(new Date(), 1) }, paymentStatus: 'PAID', } }) if (recentPayments > 5) { signals.push({ type: 'velocity', score: 50, reason: '1 цагт 5+ гүйлгээ' }) } // Signal 4: Large amount for new account const accountAge = await this.getAccountAge(userId) const amount = context.amount as number if (accountAge < 7 * 24 * 60 * 60 && amount > 100_000) { signals.push({ type: 'new_account_large_txn', score: 40, reason: 'Шинэ бүртгэл, их дүн' }) } const totalScore = Math.min(100, signals.reduce((s, sig) => s + sig.score, 0)) return { score: totalScore, action: totalScore >= 70 ? 'BLOCK' : totalScore >= 40 ? 'CHALLENGE' // Step-up OTP шаардах : 'ALLOW', signals, requiresReview: totalScore >= 70, } } } // Usage in payment flow: const risk = await fraudDetector.assess(userId, 'payment.create', { deviceId, ip, amount }) if (risk.action === 'BLOCK') { await notifyUser(userId, 'SUSPICIOUS_ACTIVITY') await securityLog(userId, 'PAYMENT_BLOCKED', risk) throw new FraudBlockError('Аюулгүй байдлын шалтгаанаар гүйлгээ түр хориглогдлоо.') } if (risk.action === 'CHALLENGE') { throw new StepUpRequiredError({ action: 'payment.create', reason: 'fraud_risk' }) } ``` ### 4. FINANCIAL DATA ENCRYPTION ```typescript // lib/security/Encryption.ts // All PII and financial data encrypted at rest const ENCRYPTION_KEY = Buffer.from(process.env.ENCRYPTION_KEY!, 'base64') // 32 bytes AES-256 export function encrypt(plaintext: string): string { const iv = crypto.randomBytes(16) const cipher = crypto.createCipheriv('aes-256-gcm', ENCRYPTION_KEY, iv) const encrypted = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()]) const tag = cipher.getAuthTag() // iv + tag + ciphertext, base64 encoded return Buffer.concat([iv, tag, encrypted]).toString('base64') } export function decrypt(ciphertext: string): string { const buf = Buffer.from(ciphertext, 'base64') const iv = buf.subarray(0, 16) const tag = buf.subarray(16, 32) const enc = buf.subarray(32) const decipher = crypto.createDecipheriv('aes-256-gcm', ENCRYPTION_KEY, iv) decipher.setAuthTag(tag) return decipher.update(enc) + decipher.final('utf8') } // Fields encrypted at rest: // - User.danRegNo (before hashing) // - SocialConnection.accessToken / pageToken // - SavedPaymentMethod.tokenRef // - BankAccount.accountNumber // - Payout.bankDetails // - Any PII fields in logs ```

ZERO-TRUST REQUEST PIPELINE ```typescript // middleware.ts — Every request goes through this pipeline export async function middleware(req: NextRequest) { const path = req.nextUrl.pathname // ── Layer 1: IP reputation check ── const ip = getClientIP(req) const isBlockedIP = await redis.get(`blocked_ip:${ip}`) if (isBlockedIP) { return new Response('Access denied', { status: 403 }) } // ── Layer 2: Bot detection ── if (isSensitivePath(path)) { const botScore = await checkBotScore(req) if (botScore > 0.8) { return NextResponse.redirect(new URL('/captcha', req.url)) } } // ── Layer 3: Rate limiting ── const rateLimitResult = await checkRateLimit(ip, path) if (rateLimitResult.exceeded) { return new Response('Rate limit exceeded', { status: 429, headers: { 'Retry-After': rateLimitResult.retryAfter.toString() } }) } // ── Layer 4: Auth verification ── if (isProtectedPath(path)) { const token = req.cookies.get('access_token')?.value if (!token) return NextResponse.redirect(new URL('/login', req.url)) try { const payload = jwt.verify(token, process.env.JWT_PUBLIC_KEY!, { algorithms: ['ES256'] }) // Attach user info to request headers for downstream use req.headers.set('x-user-id', payload.sub as string) req.headers.set('x-user-role', payload.role as string) } catch { // Token expired — redirect to refresh return NextResponse.redirect(new URL('/api/auth/refresh', req.url)) } } // ── Layer 5: CSRF protection ── if (['POST', 'PUT', 'PATCH', 'DELETE'].includes(req.method)) { const csrfToken = req.headers.get('x-csrf-token') const sessionCsrf = req.cookies.get('csrf_token')?.value if (!csrfToken || csrfToken !== sessionCsrf) { return new Response('CSRF token mismatch', { status: 403 }) } } return NextResponse.next() } ``` ### 2.
